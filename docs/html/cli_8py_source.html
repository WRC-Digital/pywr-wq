112@cli.command('run-WQ')
113@click.argument('filename', type=click.Path(dir_okay=False))
114
115def run_WQ(filename):
116 """!
117 Runs the water quality model. Requires a Pywr model stipulated as a JSON file. Also
118 requires a JSON file in which water quality parameters are stipulated for nodes, given the
119 same name as the Pywr JSON file with the suffix "_WQ". For example, if the filename parameter
120 is "test_catchment.json", the json with water quality parameters would be "test_catchment_WQ.json"
121 @param filename - the name of the pywr JSON file
122 """
123 start = timeit.default_timer()
124 base = filename[:filename.find('.')]
125 with open(filename) as fh:
126 data = json.load(fh)
127 with open(base + "_WQ.json") as fh:
128 data_WQ = json.load(fh)
129 flow_model = Model.load(data, solver='glpk-edge')
130 logger = logging.getLogger("")
131 wq_model = WQ_model.WQ_Model()
132 #The node order is set to stipulate the order in which water quailty loads are routed down the system. Only link and reservoir nodes
133 #need to be considered, i.e., no need to include abstractions and inflow nodes. This solution is not ideal as it is not a generic solution.
134 node_order = {"Node_34": 0, "Node_372": 0, "Node_7004": 0, "Node_373": 0, "DD1": 0, "DD1_release": 0,
135 "DD1_spill": 0, "Node_287": 0, "Node_100": 0, "Node_374": 0, "DD2": 0, "DD2_spill": 0,
136 "DD2_release": 0, "Node_7005": 0, "Node_392": 0, "Node_402": 0, "Grootdraai_Dam": 0,
137 "Groot_spill": 0, "Groot_release": 0, "Node_412": 0}
138 #storage nodes are stipulated as Pywr needs to know whether to pass volume or flow to the water quality model.
139 #this solution is also not ideal as it is not generic and needs to be changed for every model application
140 storage_nodes = ["DD1", "DD2", "Grootdraai_Dam"]
141 integrated_Model = flow_WQ_integration.WQ_Flow_Integration(flow_model, wq_model, data_WQ, node_order, storage_nodes)
142 integrated_Model.run()
143 #saves flow data according to recorders set in the Pywr model JSON. Water quality outputs are saved in a different way
144 with pandas.HDFStore(os.path.join('results', f"{base}_dataframes_results.h5"), mode='w') as store:
145 for rec in flow_model.recorders:
146 if hasattr(rec, 'to_dataframe'):
147 df = rec.to_dataframe()
148 store[rec.name] = df
149
150
151 stop = timeit.default_timer()
152
153 print('Time: ', stop - start)
