3def baseflow_separation(alpha, beta, flow_current, flow_before, sep_flow_before):
4 """!
5 Separate baseflow from total flow using a statistical baseflow separation method with two parameters.
6 Refer to Hughes et al. (2003) Continuous baseflow separation from time series of
7 daily and monthly streamflow data. Water SA, 29 (1).
8 @param alpha - first parameter for baseflow separation
9 @param beta - second parameter for baseflow separation
10 @param flow_current - total flow of the current timestep
11 @param flow_before - total flow of the previous timestep
12 @param sep_flow_before - baseflow of the previous timestep
13 @return baseflow of the current timestep
14 """
15 return (alpha * sep_flow_before) + beta * (1 + alpha) * (flow_current - flow_before)
